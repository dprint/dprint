== should format when using no named exports ==
export {};

[expect]
export {};

== should format when using named exports ==
const test = 5;
export {test};

[expect]
const test = 5;
export { test };

== should format when using a namespace export specifier ==
export  *   as   ns   from   "package"  ;

[expect]
export * as ns from "package";

== should format a type-only export ==
export   type    { test };

[expect]
export type { test };
