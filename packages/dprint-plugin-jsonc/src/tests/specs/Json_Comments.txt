== should format with comments ==
// test
{ // here
    // testing
    /* a */ // b
    /* test *//* b */
    "test": /* a */ "prop",
    "other": [ // test
        1234, // 1
        5678 // 2
    ], // test
    "asdf": { // test
    }
    // trailing
} // test
// testing

[expect]
// test
{
    // here
    // testing
    /* a */ // b
    /* test */ /* b */
    "test": /* a */ "prop",
    "other": [ // test
        1234, // 1
        5678 // 2
    ], // test
    "asdf": { // test
    }
    // trailing
} // test
// testing
