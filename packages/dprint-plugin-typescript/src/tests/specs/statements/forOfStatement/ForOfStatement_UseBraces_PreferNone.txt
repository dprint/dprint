== should not use braces when only one line ==
for (const t of test) {
    a;
}

[expect]
for (const t of test)
    a;

== should use braces when condition is hanging ==
for (const someLongTextThat of willBringIt)
    a;

[expect]
for (const someLongTextThat
    of willBringIt)
{
    a;
}
